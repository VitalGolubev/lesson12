Git steps:
```bash

# Init repository
git init

# Check  Status
git status

# Add(stage) files
git add .

# Commit files
git commit -m "Message for commit"

# Add remote server (option)
git remote origin git@github.com:VitalGolubev/lesson12.git

# Push to remote server (First time)
git push -u origin master

# Push files to remote server
git push

# clone repository
git clone 

# Pull from server
git pull


```
